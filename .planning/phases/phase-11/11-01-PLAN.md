    ```dart
    /// Period selection for smart plug analytics.
    enum AnalyticsPeriod { monthly, yearly, custom }

    /// A single slice of data for rendering in a pie chart.
    class PieSliceData {

      final double value;
      final double percentage;
      final Color color;
      const PieSliceData({required this.label, required this.value, required this.percentage, required this.color});

    /// Per-plug consumption breakdown.
    class PlugConsumption {
      final int plugId;
      final String plugName;
      final String roomName;
      final double consumption;
      final Color color;
      const PlugConsumption({required this.plugId, required this.plugName, required this.roomName, required this.consumption, required this.color});
    }

    /// Per-room consumption breakdown.
    class RoomConsumption {
      final int roomId;
      final String roomName;
      final double consumption;
      final Color color;
      const RoomConsumption({required this.roomId, required this.roomName, required this.consumption, required this.color});
    }

    /// Complete data package for the smart plug analytics screen.
    class SmartPlugAnalyticsData {
      final List<PlugConsumption> byPlug;
      final List<RoomConsumption> byRoom;
      final double totalSmartPlug;
      final double? totalElectricity; // null if no electricity readings
      final double? otherConsumption; // null if totalElectricity is null; clamped to >= 0
      final String unit; // always 'kWh'
      const SmartPlugAnalyticsData({
        required this.byPlug,
        required this.byRoom,
        required this.totalSmartPlug,
        this.totalElectricity,
        this.otherConsumption,
        this.unit = 'kWh',
      });
    }
    ```

    **Step 2: Create SmartPlugAnalyticsProvider**

    Create `lib/providers/smart_plug_analytics_provider.dart`:

    - Extend `ChangeNotifier`, following the exact pattern of `AnalyticsProvider` constructor style.
    - Constructor takes: `SmartPlugDao`, `ElectricityDao`, `RoomDao`, `InterpolationService`, `InterpolationSettingsProvider`.
    - State fields: `_householdId`, `_data` (SmartPlugAnalyticsData?), `_isLoading`, `_period` (AnalyticsPeriod, default monthly), `_selectedMonth` (DateTime, 1st of current month), `_selectedYear` (int, current year), `_customRange` (DateTimeRange?).
    - Getters for all state fields.
    - `setHouseholdId(int? id)` -- sets ID, clears data if null, calls `loadData()` if non-null.
    - `setPeriod(AnalyticsPeriod)` -- sets period, notifies, reloads.
    - `setSelectedMonth(DateTime)` -- normalize to 1st, clear customRange, notify, reload.
    - `navigateMonth(int delta)` -- adjust month, clear customRange, notify, reload.
    - `setSelectedYear(int)` -- set year, notify, reload.
    - `navigateYear(int delta)` -- adjust year, notify, reload.
    - `setCustomRange(DateTimeRange?)` -- set range, notify, reload.
    - `loadData()` -- the core method:
      1. Guard: if `_householdId == null`, return.
      2. Set `_isLoading = true`, notify.
      3. Compute `from`/`to` based on `_period`: monthly uses `_selectedMonth` to `_selectedMonth + 1 month`; yearly uses `Jan 1 _selectedYear` to `Jan 1 _selectedYear+1`; custom uses `_customRange!.start`/`_customRange!.end` (guard if null).
      4. Get all plugs for household via `_smartPlugDao.getSmartPlugsForHousehold(householdId)`.
      5. Get all rooms via `_roomDao.getRoomsForHousehold(householdId)`.
      6. Build a `Map<int, Room>` (roomId -> Room) for quick lookup.
      7. For each plug: call `_smartPlugDao.getTotalConsumptionForPlug(plug.id, from, to)`. Assign colors from `pieChartColors[index % pieChartColors.length]`. Create `PlugConsumption` with plug name + room name from room map.
      8. For each room: call `_smartPlugDao.getTotalConsumptionForRoom(room.id, from, to)`. Assign colors from `pieChartColors[index % pieChartColors.length]`. Create `RoomConsumption`.
      9. Total smart plug: `_smartPlugDao.getTotalSmartPlugConsumption(householdId, from, to)`.
      10. Total electricity: Use `_electricityDao.getReadingsForRange(householdId, from, to)`, convert with `fromElectricityReadings()`, then `_interpolationService.getMonthlyConsumption(...)`, sum the period consumptions. If no readings, totalElectricity = null.
      11. Other = totalElectricity != null ? max(0, totalElectricity - totalSmartPlug) : null.
      12. Build `SmartPlugAnalyticsData`, set `_data`, `_isLoading = false`, notify.
      13. Wrap entire body in try/catch. On error: `_data = null`, `_isLoading = false`, notify.
    - Import `reading_converters.dart` for `fromElectricityReadings`.
    - Use `InterpolationSettingsProvider.getMethodForMeterType('electricity')` for the interpolation method.

    ```dart
    import 'package:fl_chart/fl_chart.dart';
    import 'package:flutter/material.dart';
    import '../../l10n/app_localizations.dart';
    import '../../services/analytics/analytics_models.dart';

    /// Reusable pie chart for consumption breakdown.
    ///
    /// Follows the same stateless-widget-with-data-input pattern
    /// as ConsumptionLineChart and MonthlyBarChart.
    class ConsumptionPieChart extends StatelessWidget {
      final List<PieSliceData> slices;
      final String unit;

      const ConsumptionPieChart({
        super.key,
        required this.slices,
        required this.unit,
      });

      @override
      Widget build(BuildContext context) {
        if (slices.isEmpty) {
          return Center(child: Text(AppLocalizations.of(context)!.noData));
        }
        return PieChart(
          PieChartData(
            sections: slices.map((s) => PieChartSectionData(
              value: s.value,
              color: s.color,
              title: '${s.percentage.toStringAsFixed(1)}%',
              radius: 80,
              titleStyle: const TextStyle(
                fontSize: 12,
                fontWeight: FontWeight.bold,
                color: Colors.white,
              ),
            )).toList(),
            centerSpaceRadius: 40,
            sectionsSpace: 2,
            startDegreeOffset: -90,
          ),
        );
      }
    }
    ```
